\subsection{Exercise~1.5}

\begin{proposition}
	\label{characterizations of equivalence of norms}
	Let~$\norm{\phdot}_1$ and~$\norm{\phdot}_2$ be two norms on a vector space~$V$ with induced topologies~$\top{T}_1$ and~$\top{T}_2$ respectively.
	The following conditions on these norms and their topologies are equivalent:
	\begin{equivalenceslist}

		\item
			\label{topology is finer}
			The topology~$\top{T}_1$ is finer than the topology~$\top{T}_2$.

		\item
			\label{large balls contain small balls everywhere}
			There exists for all~$v ‚àà V$ and~$Œµ > 0$ some~$Œ¥ > 0$ with~$\ball_1(v, Œ¥) ‚äÜ \ball_2(v, Œµ)$.

		\item
			\label{large balls contain small balls}
			There exists for every~$Œµ > 0$ some~$Œ¥ > 0$ with~$\ball_1(0, Œ¥) ‚äÜ \ball_2(0, Œµ)$.

		\item
			\label{some large ball contains a small ball}
			There exists some~$Œµ, Œ¥ > 0$ with~$\ball_1(0, Œ¥) ‚äÜ \ball_2(0, Œµ)$.

		\item
			\label{there exists a constant}
			There exists a constant~$C > 0$ with~$\norm{\phdot}_2 ‚â§ C \norm{\phdot}_1$.

	\end{equivalenceslist}
\end{proposition}

\begin{proof}
	We show a bunch of implications.
	\begin{implicationslist}

		\item[\ref{topology is finer}~$\implies$~\ref{large balls contain small balls everywhere}]
			The ball~$\ball_2(v, Œµ)$ is open with respect to~$\top{T}_2$ and therefore also open with respect to~$\top{T}_1$.
			This means that~$\ball_2(v, Œµ)$ contains an open ball with respect to~$\norm{\phdot}_1$ around~$v$.

		\item[\ref{large balls contain small balls everywhere}~$\implies$~\ref{large balls contain small balls}]
			We choose~$v$ as~$0$.

		\item[\ref{large balls contain small balls}~$\implies$~\ref{some large ball contains a small ball}]
			We arbitrarily choose some~$Œµ$.

		\item[\ref{some large ball contains a small ball}~$\implies$~\ref{there exists a constant}]
			Let~$v$ be any nonzero element of~$V$ and let
			\[
				Œª ‚âî \frac{Œ¥}{2 \norm{v}_1} \,.
			\]
			Then~$\norm{Œª v}_1 = Œ¥ / 2$, and therefore~$\norm{Œª v}_2 < Œµ$.
			But
			\[
				\norm{Œª v}_2
				=
				\frac{ Œ¥ \norm{v}_2 }{ 2 \norm{v}_1 } \,.
			\]
			The inequality~$\norm{Œª v}_2 < Œµ$ can therefore be rearranged to
			\[
				\norm{v}_2 < \frac{2 Œµ}{Œ¥} \norm{v}_1 \,.
			\]
			We may therefore choose the required constant~$C$ as~$2 Œµ / Œ¥$.

		\item[\ref{there exists a constant}~$\implies$~\ref{large balls contain small balls}]
			We may choose~$Œ¥$ as~$Œµ / C$.

		\item[\ref{large balls contain small balls}~$\implies$~\ref{large balls contain small balls everywhere}]
			This implication follows from~$V \to V$,~$x \mapsto x + v$ being an isometry.

		\item[\ref{large balls contain small balls everywhere}~$\implies$~\ref{topology is finer}]
			Let~$U$ be a subset of~$V$ that is open with respect to~$\top{T}_2$.
			Let~$v$ be a point in~$U$.
			There exists some radius~$Œµ > 0$ with~$\ball_2(v, Œµ) ‚äÜ U$.
			It follows that there exists some radius~$Œ¥ > 0$ with~$\ball_1(v, Œ¥) ‚äÜ \ball_2(v, Œµ)$.
			This shows that~$U$ is also open with respect to~$\top{T}_1$.
		\qedhere

	\end{implicationslist}
\end{proof}

Let now~$ùïÇ$ be any of the topological skew fields~$‚Ñù$,~$‚ÑÇ$ or~$‚Ñç$.
We will show that all norm on a finite-dimensional~\vectorspace{$ùïÇ$} induce the same topology.
It suffices to show that all norms on~$ùïÇ^n$ induce the same topology.
We will thus show that all norms on~$ùïÇ^n$ induce the standard topology (i.e., the topology induced by~$\norm{\phdot}_2$.)

We first show that all norms~$\norm{\phdot}_p$ with~$1 ‚â§ p ‚â§ ‚àû$ induce the standard topology, which is true for~$p = 2$ by definition.
Let~$1 ‚â§ p < ‚àû$.
We observe on the one hand the inequality
\[
	\norm{x}_p
	=
	\biggl( ‚àë_{i = 1}^n \abs{x_i}^p \biggr)^{1 / p}
	‚â§
	\biggl( ‚àë_{i = 1}^n \norm{x}_‚àû^p \biggr)^{1 / p}
	=
	( n \norm{x}_‚àû^p )^{1 / p}
	=
	n^{1 / p} \norm{x}_‚àû \,.
\]
For every index~$j = 1, \dotsc, n$ we have~$\abs{x_j} ‚â§ ( \sum_{i = 1}^n \abs{x_i}^p )^{1 / p}$, and therefore on the other hand the inequality
\[
	\norm{x}_‚àû
	=
	\max( \abs{x_1}, \dotsc, \abs{x_n} )
	‚â§
	\biggl( \sum_{i = 1}^n \abs{x_i}^p \biggr)^{1 / p}
	=
	\norm{x}_p \,.
\]
According to \cref{characterizations of equivalence of norms}, the above two inequalities tell us that the two norms~$\norm{\phdot}_p$ and~$\norm{\phdot}_‚àû$ induce the same topology, for every~$1 ‚â§ p < ‚àû$.
This shows that~$\norm{\phdot}_‚àû$ induces the standard topology (by considering the case~$p = 2$) and then furthermore that each~$\norm{\phdot}_p$ induces the standard topology.

\begin{lemma}
	\label{norm is continuous}
	Let~$V$ be a vector space and let~$\norm{\phdot}$ be a norm on~$V$.
	The norm~$\norm{\phdot}$ is continuous as a map from~$V$ to~$‚Ñù$ (with respect to the standard topology on~$‚Ñù$ and the topology on~$V$ induced by~$\norm{\phdot}$).
\end{lemma}

\begin{proof}
	This is a special case of Example~1.13 by choosing~$x = 0$.
\end{proof}

Let now~$\norm{\phdot}$ be an arbitrary norm on~$ùïÇ^n$.
We show that the topology induced by~$\norm{\phdot}$ is both coarser and finer than the standard topology.

For~$C ‚âî \max(\norm{e_1}, \dotsc, \norm{e_n})$ we have
\[
	\norm{x}
	=
	\norm[\Bigg]{ ‚àë_{i = 1}^n x_i e_i }
	‚â§
	‚àë_{i = 1}^n \abs{x_i} \norm{e_i}
	‚â§
	C ‚àë_{i = 1}^n \abs{x_i}
	=
	C \norm{x}_1 \,.
\]
This shows that the topology induced by~$\norm{\phdot}$ is coarser than the topology induced by~$\norm{\phdot}_1$, i.e., coarser than the standard topology.
We note that by \cref{norm is continuous} this entails that the norm~$\norm{\phdot}$ is continuous with respect to the standard topologies on~$ùïÇ^n$ and~$‚Ñù$.

On the other hand we consider the closed unit ball
\[
	S ‚âî \{ x ‚àà ùïÇ^n \suchthat \norm{x}_2 ‚â§ 1 \} \,.
\]
This subspace of~$ùïÇ^n$ is compact (since it is homeomorphic to the closed unit ball in~$‚Ñù^m$ for~$m = \dim_‚Ñù ùïÇ ‚ãÖ n$).
It follows from this compactness and the continuity of~$\norm{\phdot}$ that~$\norm{\phdot}$ is bounded on~$S$.
There hence exists some~$C' > 0$ with~$\norm{x} < C'$ for every~$x ‚àà S$.
This tells us that
\[
	\{ x ‚àà ùïÇ^n \suchthat \norm{x}_2 < 1 \}
	‚äÜ
	\{ x ‚àà ùïÇ^n \suchthat \norm{x} < C' \} \,.
\]
According to \cref{characterizations of equivalence of norms} this shows that the topology induced by~$\norm{\phdot}$ is finer than the standard topology.
